worker_processes 1;

error_log logs/error.log warn;

events { worker_connections 1024; }

http {
    sendfile on;
    log_format main 	'$remote_addr - $remote_user [$time_local]  $status '
    			'"$request" [$content_type] {$request_body} $body_bytes_sent "$http_referer" '
    			'"$http_user_agent" "$http_x_forwarded_for"';

    upstream front-end {
        server us4-fe:80;
    }
 
    upstream back-end {
        server us4-be:8080;
    }

    server {
	listen       80;
	server_name  intproj21.sit.kmutt.ac.th;
        access_log /spool/logs/nginx-access.log main;
	
	location / {
	    add_header Access-Control-Allow-Origin "https://intproj21.sit.kmutt.ac.th";
	    proxy_pass http://front-end;
	    proxy_redirect     off;
	    proxy_set_header   Host $host;
	    proxy_set_header   X-Real-IP $remote_addr;
	    proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
	    proxy_set_header   X-Forwarded-Host $server_name;
	}

	location /api {
	    add_header Access-Control-Allow-Origin "https://intproj21.sit.kmutt.ac.th";
	    proxy_pass http://back-end;
	    proxy_redirect     off;
	    proxy_set_header   Host $host;
	    proxy_set_header   X-Real-IP $remote_addr;
	    proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
	    proxy_set_header   X-Forwarded-Host $server_name;
	}
    }
}
